🧠 models/deepseek_coder.py

Pure code engine

class DeepSeekCoder:
    def __init__(self, model):
        self.model = model

    def run(self, task: str) -> str:
        prompt = f"""
You are a professional software engineer.

Rules:
- Output ONLY code or technical explanations
- NO conversation
- NO persona
- NO emojis
- Be correct and explicit

Task:
{task}
"""
        res = self.model(prompt, max_tokens=1024)
        return res["choices"][0]["text"].strip()

🔀 core/router.py (UPDATED)

Now supports multiple specialists.

class TaskRouter:
    def __init__(self, reasoning_model, coder_model):
        self.reasoning = reasoning_model
        self.coder = coder_model

    def execute(self, plan, user_input):
        outputs = {}

        if plan.reasoning_required:
            outputs["reasoning"] = self.reasoning.run(user_input)

        if plan.coding_required:
            outputs["code"] = self.coder.run(user_input)

        return outputs
