🧠 memory/importance.py

Decides what is worth remembering

class ImportanceScorer:
    def score(self, text: str) -> float:
        """
        0.0–1.0 scale
        """
        keywords = [
            "always", "never", "prefer", "my system",
            "important", "remember", "from now on"
        ]

        score = 0.2
        for k in keywords:
            if k in text.lower():
                score += 0.15

        return min(score, 1.0)
🧠 memory/summarizer.py

Compress conversations into episodes

class ConversationSummarizer:
    def __init__(self, model):
        self.model = model

    def summarize(self, dialogue: str) -> str:
        prompt = f"""
Summarize the following conversation into
a short factual memory.

Rules:
- No dialogue
- No opinions
- No fluff
- Max 5 bullet points

Conversation:
{dialogue}
"""
        res = self.model(prompt, max_tokens=256)
        return res["choices"][0]["text"].strip()
🧠 memory/decay.py

Forgetting logic

import time

class MemoryDecay:
    def __init__(self, half_life_days=30):
        self.half_life = half_life_days * 86400

    def decay_factor(self, timestamp: float) -> float:
        age = time.time() - timestamp
        return 0.5 ** (age / self.half_life)

Now stores metadata.

class LongTermMemory:
    def __init__(self, dim=1024):
        self.index = faiss.IndexFlatL2(dim)
        self.records = []

    def add(self, embedding, text, importance):
        self.index.add(np.array([embedding]))
        self.records.append({
            "text": text,
            "importance": importance,
            "timestamp": time.time()
        })

    def query(self, embedding, k=5, decay=None):
        if self.index.ntotal == 0:
            return []

        _, ids = self.index.search(np.array([embedding]), k)
        results = []

        for i in ids[0]:
            rec = self.records[i]
            score = rec["importance"]

            if decay:
                score *= decay.decay_factor(rec["timestamp"])

            if score > 0.15:
                results.append(rec["text"])

        return results
🧠 memory/episodic.py (UPDATED)
class EpisodicMemory:
    def __init__(self):
        self.episodes = []

    def add(self, summary: str):
        self.episodes.append({
            "summary": summary,
            "timestamp": time.time()
        })

    def recent(self, n=3):
        return [e["summary"] for e in self.episodes[-n:]]
